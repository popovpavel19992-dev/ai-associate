# 4.7 Opposing-Counsel Response Predictor — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Predict how opposing counsel will respond to a specific filing (motion / demand letter / discovery set) and surface a general posture readout for the attorney across the case.

**Architecture:** Mirror Phase 4 pipeline shape (cache → credits → identify → enrich → sources → Claude → tx insert). Identity rides on existing `case_parties` (`role='opposing_counsel'`); a new `opposing_counsel_profiles` row maps that party to a CourtListener person. Two new prediction tables: `opposing_counsel_predictions` (filing-specific) and `opposing_counsel_postures` (general). UI: inline button on three drafters + new "Opposing Counsel" tab on case page. Auto-extract from uploaded opposing documents via Claude haiku.

**Tech Stack:** TypeScript, Drizzle ORM, tRPC v11, Vitest, Playwright, Claude (haiku-4-5 + sonnet-4-6), Voyage embeddings, CourtListener REST v4, Next.js App Router, React 19.

**Spec:** `docs/superpowers/specs/2026-05-01-4.7-opposing-counsel-response-predictor-design.md`

**Beta gate:** existing `STRATEGY_BETA_ORG_IDS` env var.

**Pricing:** 2 credits flat per `predictResponse` cache miss; 2 credits flat per `getPosture` cache miss; 0 credits on cache hit; 0 credits on auto-extract (haiku is treated as housekeeping). Refund on Claude error.

```ts
// src/server/services/opposing-counsel/identify.ts
import { getCourtListenerClient } from "@/server/services/courtlistener/client";
import type { PeoplePerson } from "@/server/services/courtlistener/types";

const SUFFIX_RE = /\b(esq\.?|jr\.?|sr\.?|iii|iv|ii)\b/gi;

export function normalizeName(raw: string): string {
  return raw
    .toLowerCase()
    .replace(SUFFIX_RE, "")
    .replace(/\b[a-z]\.\s/g, "") // middle initials
    .replace(/[^\p{L}\s]/gu, "")
    .replace(/\s+/g, " ")
    .trim();
}

function lastTokenScore(a: string, b: string): number {
  const al = a.split(" ").pop() ?? "";
  const bl = b.split(" ").pop() ?? "";
  return al && al === bl ? 0.5 : 0;
}

function fullExactScore(a: string, b: string): number {
  return a === b ? 0.5 : 0;
}

function firmBoost(firm: string | undefined, person: PeoplePerson): number {
  if (!firm) return 0;
  const norm = firm.toLowerCase().trim();
  return (person.positions ?? []).some((p) =>
    p.organization_name?.toLowerCase().includes(norm),
  )
    ? 0.2
    : 0;
}

export interface MatchResult {
  clPersonId: string;
  clFirmName: string | null;
  confidence: number;
}

export async function matchAttorney(args: {
  name: string;
  firm?: string;
}): Promise<MatchResult | null> {
  const target = normalizeName(args.name);
  if (!target) return null;
  const client = getCourtListenerClient();
  let res;
  try {
    res = await client.people({ name: args.name, pageSize: 10 });
  } catch {
    return null;
  }
  let best: { person: PeoplePerson; score: number } | null = null;
  for (const p of res.results) {
    const cand = normalizeName(p.name_full);
    const score = fullExactScore(target, cand) + lastTokenScore(target, cand) + firmBoost(args.firm, p);
    if (!best || score > best.score) best = { person: p, score };
  }
  if (!best || best.score < 0.6) return null;
  const firmName =
    best.person.positions?.find((p) => p.organization_name)?.organization_name ?? null;
  return {
    clPersonId: String(best.person.id),
    clFirmName: firmName,
    confidence: Math.min(1, best.score),
  };
}
```

```ts
// src/server/services/opposing-counsel/enrich.ts
import { getCourtListenerClient } from "@/server/services/courtlistener/client";

const STALE_MS = 7 * 24 * 3600 * 1000;

export interface EnrichmentJson {
  totalFilings: number;
  motionMix: { label: string; count: number; pct: number }[];
  earliest?: string;
  latest?: string;
  fetchedAt: string;
}

export function isStale(fetchedAt: Date | null): boolean {
  if (!fetchedAt) return true;
  return Date.now() - fetchedAt.getTime() > STALE_MS;
}

const MOTION_BUCKETS = [
  "Motion to Dismiss",
  "Motion for Summary Judgment",
  "Motion to Compel",
  "Motion in Limine",
  "Motion for Sanctions",
];

function bucketize(description: string): string | null {
  const d = description.toLowerCase();
  for (const b of MOTION_BUCKETS) if (d.includes(b.toLowerCase())) return b;
  return null;
}

export async function fetchEnrichment(args: {
  clPersonId: string;
}): Promise<EnrichmentJson | null> {
  const client = getCourtListenerClient();
  try {
    const res = await client.search({
      q: `attorney_id:${args.clPersonId}`,
      type: "r",
      pageSize: 50,
    } as never);
    const counts = new Map<string, number>();
    let earliest: string | undefined;
    let latest: string | undefined;
    for (const r of res.results as Array<{ description?: string; date_filed?: string }>) {
      const b = bucketize(r.description ?? "");
      if (b) counts.set(b, (counts.get(b) ?? 0) + 1);
      if (r.date_filed) {
        if (!earliest || r.date_filed < earliest) earliest = r.date_filed;
        if (!latest || r.date_filed > latest) latest = r.date_filed;
      }
    }
    const total = res.results.length;
    const motionMix = [...counts.entries()]
      .map(([label, count]) => ({ label, count, pct: total ? count / total : 0 }))
      .sort((a, b) => b.count - a.count);
    return {
      totalFilings: total,
      motionMix,
      earliest,
      latest,
```

```ts
function assertBetaOrg(orgId: string) {
  const allowed = (process.env.STRATEGY_BETA_ORG_IDS ?? "")
    .split(",").map((s) => s.trim()).filter(Boolean);

function sha(input: string) {
  return createHash("sha256").update(input).digest("hex");
}

async function resolveProfile(args: {
  orgId: string;
  caseId: string;
  profileId?: string;
}): Promise<OpposingCounselProfile> {
  // List all opposing-counsel profiles for this case (org-scoped)
  const rows = await db
    .select({
      profile: opposingCounselProfiles,
      party: caseParties,
    })
    .from(opposingCounselProfiles)
    .innerJoin(caseParties, eq(caseParties.id, opposingCounselProfiles.casePartyId))
    .where(and(
      eq(opposingCounselProfiles.orgId, args.orgId),
      eq(caseParties.caseId, args.caseId),
      eq(caseParties.role, "opposing_counsel"),
    ));

  if (args.profileId) {
    const m = rows.find((r) => r.profile.id === args.profileId);
    if (!m) throw new NeedsAttorneyError();
    return m.profile;
  }
  if (rows.length === 0) throw new NeedsAttorneyError();
  if (rows.length > 1) {
    throw new NeedsAttorneyChoiceError(rows.map((r) => ({
      profileId: r.profile.id, name: r.party.name, firm: r.profile.clFirmName,
    })));
  }
  return rows[0].profile;
}

  if (hit) return hit;

  const credit = await decrementCredits({
    orgId: args.orgId, userId: args.userId, amount: COST_PREDICT, reason: "opposing_counsel.predict",
  });
  if (!credit.ok) throw new InsufficientCreditsError();

  try {
    const matched = await ensureCLMatch(profile, name);
    const enriched = await ensureEnrichment(matched);
    const sources = await collectFilingSources({
      caseId: args.caseId,
      query: `${args.targetTitle}\n${args.targetBody}`.slice(0, 1000),
      k: 5,
    });
    const result = await runPrediction({
      target: { kind: args.targetKind, title: args.targetTitle, body: args.targetBody },
      profile: { name, firm: enriched.clFirmName, clMatched: !!enriched.clPersonId },
      enrichment: (enriched.enrichmentJson as never) ?? null,
      sources,
    });

    const [row] = await db.insert(opposingCounselPredictions).values({
      orgId: args.orgId,
      caseId: args.caseId,
      profileId: enriched.id,
      targetKind: args.targetKind,
      targetId: args.targetId,
      cacheHash,
      likelyResponse: result.likelyResponse,
      keyObjections: result.keyObjections,
      settleProbLow: String(result.settleProbLow) as never,
      settleProbHigh: String(result.settleProbHigh) as never,
      estResponseDaysLow: result.estResponseDaysLow,
      estResponseDaysHigh: result.estResponseDaysHigh,
      aggressiveness: result.aggressiveness,
      recommendedPrep: result.recommendedPrep,
      reasoningMd: result.reasoningMd,
      sourcesJson: result.sources,
      confidenceOverall: result.confidenceOverall,
    }).returning();

    return row;
  } catch (e) {
    await refundCredits({
      orgId: args.orgId, userId: args.userId, amount: COST_PREDICT, eventId: credit.eventId,
    });
    throw e;
  }
}

export interface PostureArgs {
  orgId: string;
  userId: string;
  caseId: string;
  profileId: string;
  regenerateSalt?: number;
}

async function caseStateHash(caseId: string): Promise<string> {
  const [row] = await db
    .select({ latest: max(caseDocuments.createdAt) })
    .from(caseDocuments)
    .where(eq(caseDocuments.caseId, caseId));
  return sha(`${caseId}:${row?.latest?.toISOString() ?? ""}`);
}

export async function getPosture(args: PostureArgs) {
  assertBetaOrg(args.orgId);
  const profile = await resolveProfile(args);
  const name = await partyName(profile.casePartyId);

  const stateHash = await caseStateHash(args.caseId);
  const cacheHash = sha([args.caseId, profile.id, stateHash, args.regenerateSalt ?? 0].join(":"));

  const [hit] = await db.select().from(opposingCounselPostures)
    .where(and(
      eq(opposingCounselPostures.orgId, args.orgId),
      eq(opposingCounselPostures.cacheHash, cacheHash),
    ));
  if (hit) return hit;

  const credit = await decrementCredits({
    orgId: args.orgId, userId: args.userId, amount: COST_POSTURE, reason: "opposing_counsel.posture",
  });
  if (!credit.ok) throw new InsufficientCreditsError();

  try {
    const matched = await ensureCLMatch(profile, name);
    const enriched = await ensureEnrichment(matched);
    const sources = await collectPostureSources({ caseId: args.caseId });
    const result = await runPosture({
      profile: { name, firm: enriched.clFirmName, clMatched: !!enriched.clPersonId },
      enrichment: (enriched.enrichmentJson as never) ?? null,
      sources,
    });
    const [row] = await db.insert(opposingCounselPostures).values({
      orgId: args.orgId,
      caseId: args.caseId,
      profileId: enriched.id,
      cacheHash,
      aggressiveness: result.aggressiveness,
      settleLow: String(result.settleLow) as never,
      settleHigh: String(result.settleHigh) as never,
      settleLikelihood: String((result.settleLow + result.settleHigh) / 2) as never,
      typicalMotions: result.typicalMotions,
      reasoningMd: result.reasoningMd,
      sourcesJson: result.sources,
      confidenceOverall: result.confidenceOverall,
    }).returning();
    return row;
  } catch (e) {
    await refundCredits({
      orgId: args.orgId, userId: args.userId, amount: COST_POSTURE, eventId: credit.eventId,
    });
    throw e;
  }
}
```

```ts
// src/server/trpc/routers/opposing-counsel.ts
import { z } from "zod";
import { TRPCError } from "@trpc/server";
import { router, protectedProcedure } from "@/server/trpc";
import {
  predictResponse,
  getPosture,
  attachAttorney,
  NotBetaOrgError,
  NeedsAttorneyError,
  NeedsAttorneyChoiceError,
  InsufficientCreditsError,
} from "@/server/services/opposing-counsel";
import { db } from "@/server/db";
import { and, eq } from "drizzle-orm";
import { caseParties } from "@/server/db/schema/case-parties";
import { opposingCounselProfiles } from "@/server/db/schema/opposing-counsel-profiles";
import { opposingCounselPredictions } from "@/server/db/schema/opposing-counsel-predictions";

function mapErr(e: unknown): never {
  if (e instanceof NotBetaOrgError) throw new TRPCError({ code: "FORBIDDEN", message: e.message });
  if (e instanceof InsufficientCreditsError) throw new TRPCError({ code: "PAYMENT_REQUIRED" as never, message: e.message });
  if (e instanceof NeedsAttorneyError) throw new TRPCError({ code: "PRECONDITION_FAILED", message: "needsAttorney" });
  if (e instanceof NeedsAttorneyChoiceError) throw new TRPCError({ code: "PRECONDITION_FAILED", message: "needsAttorneyChoice", cause: { options: e.options } });
  throw e;
}

export const opposingCounselRouter = router({
  predictResponse: protectedProcedure
    .input(z.object({
      caseId: z.string().uuid(),
      targetKind: z.enum(["motion", "demand_letter", "discovery_set"]),
      targetId: z.string().uuid(),
      targetTitle: z.string().min(1),
      targetBody: z.string().min(1),
      profileId: z.string().uuid().optional(),
      regenerate: z.boolean().optional(),
    }))
    .mutation(async ({ ctx, input }) => {
      try {
        return await predictResponse({
          orgId: ctx.orgId,
          userId: ctx.userId,
          caseId: input.caseId,
          targetKind: input.targetKind,
          targetId: input.targetId,
          targetTitle: input.targetTitle,
          targetBody: input.targetBody,
          profileId: input.profileId,
          regenerateSalt: input.regenerate ? Date.now() : undefined,
        });
      } catch (e) { mapErr(e); }
    }),

  getPosture: protectedProcedure
    .input(z.object({
      caseId: z.string().uuid(),
      profileId: z.string().uuid(),
      regenerate: z.boolean().optional(),
    }))
    .mutation(async ({ ctx, input }) => {
      try {
        return await getPosture({
          orgId: ctx.orgId,
          userId: ctx.userId,
          caseId: input.caseId,
          profileId: input.profileId,
          regenerateSalt: input.regenerate ? Date.now() : undefined,
        });
      } catch (e) { mapErr(e); }
    }),

  attachAttorney: protectedProcedure
    .input(z.object({
      caseId: z.string().uuid(),
      casePartyId: z.string().uuid(),
      firm: z.string().nullable().optional(),
      barNumber: z.string().nullable().optional(),
      barState: z.string().length(2).nullable().optional(),
    }))
    .mutation(async ({ ctx, input }) => {
      try {
        return await attachAttorney({
          orgId: ctx.orgId, userId: ctx.userId, caseId: input.caseId,
          casePartyId: input.casePartyId, firm: input.firm, barNumber: input.barNumber, barState: input.barState,
        });
      } catch (e) { mapErr(e); }
    }),

  listAttorneysForCase: protectedProcedure
    .input(z.object({ caseId: z.string().uuid() }))
    .query(async ({ ctx, input }) => {
      return await db
        .select({
          profile: opposingCounselProfiles,
          party: caseParties,
        })
        .from(caseParties)
        .leftJoin(opposingCounselProfiles, eq(opposingCounselProfiles.casePartyId, caseParties.id))
        .where(and(
          eq(caseParties.orgId, ctx.orgId),
          eq(caseParties.caseId, input.caseId),
          eq(caseParties.role, "opposing_counsel"),
        ));
    }),

  listPredictionsForTarget: protectedProcedure
    .input(z.object({
      caseId: z.string().uuid(),
      targetKind: z.enum(["motion", "demand_letter", "discovery_set"]),
      targetId: z.string().uuid(),
    }))
    .query(async ({ ctx, input }) => {
      return await db.select().from(opposingCounselPredictions)
        .where(and(
          eq(opposingCounselPredictions.orgId, ctx.orgId),
          eq(opposingCounselPredictions.caseId, input.caseId),
          eq(opposingCounselPredictions.targetKind, input.targetKind),
          eq(opposingCounselPredictions.targetId, input.targetId),
        ))
        .orderBy(opposingCounselPredictions.createdAt);
    }),
});
```

```tsx
import type { PredictionTargetKind } from "@/server/db/schema/opposing-counsel-predictions";

interface Props {
  open: boolean;
  onOpenChange: (v: boolean) => void;
  caseId: string;
  targetKind: PredictionTargetKind;
  targetId: string;
  targetTitle: string;
  targetBody: string;

export function PredictionDialog(props: Props) {
  const [profileId, setProfileId] = useState<string | undefined>();
  const [choices, setChoices] = useState<Array<{ profileId: string; name: string; firm?: string | null }> | null>(null);

  const predict = trpc.opposingCounsel.predictResponse.useMutation({
    onError: (e) => {

      targetKind: props.targetKind,
      targetId: props.targetId,
      targetTitle: props.targetTitle,
      targetBody: props.targetBody,
```

```tsx
"use client";
import { useState } from "react";
import { Dialog, DialogContent, DialogHeader, DialogTitle } from "@/components/ui/dialog";
import { Input } from "@/components/ui/input";
import { Button } from "@/components/ui/button";
import { trpc } from "@/lib/trpc";

export function AttorneyAttachDialog(props: { open: boolean; onOpenChange: (v: boolean) => void; caseId: string; onAdded?: () => void }) {
  const [name, setName] = useState("");
  const [firm, setFirm] = useState("");
  const [bar, setBar] = useState("");
  const [state, setState] = useState("");
  const createParty = trpc.cases.parties.create.useMutation();
  const attach = trpc.opposingCounsel.attachAttorney.useMutation();

  async function submit() {
    const party = await createParty.mutateAsync({ caseId: props.caseId, role: "opposing_counsel", name });
    await attach.mutateAsync({
      caseId: props.caseId, casePartyId: party.id,
      firm: firm || null, barNumber: bar || null, barState: state || null,
    });
    props.onAdded?.();
    props.onOpenChange(false);
  }

  return (
    <Dialog open={props.open} onOpenChange={props.onOpenChange}>
      <DialogContent>
        <DialogHeader><DialogTitle>Add opposing counsel</DialogTitle></DialogHeader>
        <div className="space-y-2">
          <Input placeholder="Attorney name" value={name} onChange={(e) => setName(e.target.value)} />
          <Input placeholder="Firm" value={firm} onChange={(e) => setFirm(e.target.value)} />
          <Input placeholder="Bar number" value={bar} onChange={(e) => setBar(e.target.value)} />
          <Input placeholder="Bar state (2-letter)" value={state} onChange={(e) => setState(e.target.value.toUpperCase())} maxLength={2} />
          <Button disabled={!name} onClick={submit}>Add</Button>
        </div>
      </DialogContent>
    </Dialog>
  );
}
```

```tsx
"use client";
import { useState } from "react";
```
